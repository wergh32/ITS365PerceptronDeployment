import numpy as np import random # --------------------- # 1️⃣ Generate training data # --------------------- def generate_L(): grid = np.zeros((4,4), dtype=int) row = random.randint(0,2) col = random.randint(0,2) vert_len = random.randint(2, 4-row) horiz_len = random.randint(2, 4-col) for r in range(row, row+vert_len): grid[r][col] = 1 for c in range(col, col+horiz_len): grid[row+vert_len-1][c] = 1 return grid def generate_T(): grid = np.zeros((4,4), dtype=int) row = random.randint(0,2) col = random.randint(0,1) horiz_len = random.randint(2, min(4-col,4)) vert_len = random.randint(2, 4-row) for c in range(col, col+horiz_len): grid[row][c] = 1 mid = col + horiz_len // 2 for r in range(row, row+vert_len): grid[r][mid] = 1 return grid # Generate 100 samples each L_shapes = [generate_L() for _ in range(100)] T_shapes = [generate_T() for _ in range(100)] X_train = np.array([l.flatten() for l in L_shapes] + [t.flatten() for t in T_shapes]) y_train = np.array([1]*100 + [-1]*100) # Shuffle indices = np.arange(len(X_train)) np.random.shuffle(indices) X_train = X_train[indices] y_train = y_train[indices] # --------------------- # 2️⃣ Train perceptron # --------------------- def fit(X_train, y_train, epochs=500, lr=0.1): n_features = X_train.shape[1] weights = np.zeros(n_features) bias = 0 for _ in range(epochs): for i, x in enumerate(X_train): y_pred = np.sign(np.dot(weights, x) + bias) if y_pred == 0: y_pred = -1 if y_train[i]*y_pred <= 0: weights += lr * y_train[i] * x bias += lr * y_train[i] return weights, bias weights, bias = fit(X_train, y_train) # --------------------- # 3️⃣ Predict function # --------------------- def predict(grid_flat): return np.sign(np.dot(weights, grid_flat) + bias) # --------------------- # 4️⃣ DOM interaction # --------------------- from js import document grid_table = document.getElementById("grid").getElementsByTagName("tbody")[0] # create 4x4 cells cells = [] for r in range(4): row = document.createElement("tr") row_cells = [] for c in range(4): td = document.createElement("td") td.onclick = lambda e, td=td: td.classList.toggle("active") row.appendChild(td) row_cells.append(td) grid_table.appendChild(row) cells.append(row_cells) def classify(*args): # read grid grid_flat = np.array([1 if td.classList.contains("active") else 0 for row in cells for td in row]) y = predict(grid_flat) result_text = "L" if y==1 else "T" document.getElementById("result").innerText = f"Result: {result_text}" document.getElementById("classify-btn").onclick = classify
